def pretty_value
  if code == 'illness_code'
    illness = Illness.find_by(code: value)
    (illness.nil?) ? "" : illness.name
  elsif code == 'symptom_codes'
    codes = JSON.parse(value)
    Symptom.where(code: codes).map(&:name).join(', ')
  else
    value
  end
end
