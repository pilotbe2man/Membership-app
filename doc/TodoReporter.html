def percentages
    set_task_completes
    set_total_number_of_todo_task_completes
    return {
        complete: @total_completes.zero? ? 0 : percent_of(pass, @total_completes),
        incomplete: @total_completes.zero? ? 0 : percent_of(failed, @total_completes)
    }
end
