def merge_path(keys, value)
   keys[0...-1].inject(self) do |result, key|
    result[key] ||= {}
  end[keys[-1]] = value
  self
end

def walk(path=[], &block)
  self.each do |key, value|
    if value.is_a?(Hash)
      value.walk(path + [key], &block)
    else
      yield [path, key, value]
    end
  end
end
